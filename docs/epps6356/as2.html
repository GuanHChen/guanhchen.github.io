# Load interpolation helper
library(akima)   # for interp()
# Take your raw data
x <- df$WB
y <- df$GDP
z <- df$LE

x_scaled <- (x - min(x)) / (max(x) - min(x))
y_scaled <- (y - min(y)) / (max(y) - min(y))

interp_grid <- interp(x_scaled, y_scaled, z, duplicate = "mean", nx = 30, ny = 30)

par(mar = c(2, 2, 2, 2), lwd = 0.5)

# 3D surface plot
